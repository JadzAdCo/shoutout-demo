rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isSelf(userId) {
      return signedIn() && request.auth.uid == userId;
    }

    function isParticipant() {
      return signedIn()
        && resource.data.participants is list
        && request.auth.uid in resource.data.participants;
    }

    function isNewParticipant() {
      return signedIn()
        && request.resource.data.participants is list
        && request.auth.uid in request.resource.data.participants;
    }

    function isMutualMingl(connectionId) {
      return signedIn()
        && exists(/databases/$(database)/documents/minglConnections/$(connectionId))
        && get(/databases/$(database)/documents/minglConnections/$(connectionId)).data.status == "mutual"
        && request.auth.uid in get(/databases/$(database)/documents/minglConnections/$(connectionId)).data.participants;
    }

    function isExistingMinglRoom() {
      return resource.data.type == "mingl";
    }

    function isNewMinglRoom() {
      return request.resource.data.type == "mingl";
    }

    function isExistingMinglMessage() {
      return resource.data.keys().hasAny(["roomType"])
        && resource.data.roomType == "mingl";
    }

    function isNewMinglMessage() {
      return request.resource.data.keys().hasAny(["roomType"])
        && request.resource.data.roomType == "mingl";
    }

    match /users/{userId} {
      allow create: if isSelf(userId);
      allow read: if signedIn();
      allow update: if isSelf(userId);
      allow delete: if false;
    }

    match /clubs/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /clubLocations/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /events/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /templates/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /shoutouts/{id} {
      allow create: if signedIn();
      allow read, update, delete: if signedIn();
    }

    match /liveContent/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /guestListRequests/{id} {
      allow create: if signedIn();
      allow read, update, delete: if signedIn();
    }

    match /messages/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /privacyConsents/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /friendRequests/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /friendships/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /notifications/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /roleRequests/{id} {
      allow create, read, update, delete: if signedIn();
    }`r`n
    match /patronRanks/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /approvedShoutOutLibrary/{id} {
      allow read: if true;
      allow create, update, delete: if signedIn();
    }

    match /translationSettings/{id} {
      allow read: if true;
      allow write: if signedIn();
    }

    match /inboxNotifications/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /shoutoutAudit/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /djProfiles/{id} {
      allow read: if true;
      allow create, update, delete: if signedIn();
    }

    match /promoterProfiles/{id} {
      allow read: if true;
      allow create, update, delete: if signedIn();
    }

    match /shoutoutRecommendations/{id} {
      allow create, read, update, delete: if signedIn();
    }

    match /minglConnections/{connectionId} {
      allow read: if isParticipant();
      allow create: if isNewParticipant()
        && request.resource.data.participants.size() == 2
        && request.auth.uid == request.resource.data.requestedBy;
      allow update: if isParticipant()
        && request.resource.data.participants == resource.data.participants
        && (
          request.auth.uid == resource.data.requestedTo ||
          request.auth.uid == resource.data.requestedBy
        );
      allow delete: if false;
    }

    match /chatRooms/{roomId} {
      allow read: if signedIn()
        && (
          isParticipant() ||
          !isExistingMinglRoom()
        );
      allow create: if signedIn()
        && (
          (
            isNewMinglRoom()
            && isNewParticipant()
            && request.resource.data.connectionId is string
            && isMutualMingl(request.resource.data.connectionId)
          ) ||
          !isNewMinglRoom()
        );
      allow update: if signedIn()
        && (
          (
            isExistingMinglRoom()
            && isParticipant()
            && resource.data.connectionId is string
            && isMutualMingl(resource.data.connectionId)
          ) ||
          !isExistingMinglRoom()
        );
      allow delete: if signedIn() && !isExistingMinglRoom();
    }

    match /chatMessages/{messageId} {
      allow read: if signedIn()
        && (
          (
            isExistingMinglMessage()
            && resource.data.participants is list
            && request.auth.uid in resource.data.participants
          ) ||
          !isExistingMinglMessage()
        );
      allow create: if signedIn()
        && (
          (
            isNewMinglMessage()
            && request.auth.uid == request.resource.data.senderUid
            && request.resource.data.participants is list
            && request.auth.uid in request.resource.data.participants
            && exists(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId))
            && get(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId)).data.type == "mingl"
            && request.auth.uid in get(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId)).data.participants
            && isMutualMingl(get(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId)).data.connectionId)
          ) ||
          !isNewMinglMessage()
        );
      allow update, delete: if signedIn()
        && !isExistingMinglMessage();
    }
  }
}
