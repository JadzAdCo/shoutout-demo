rules_version = '2';

service cloud.firestore {
  match /databases/{database}/documents {
    function signedIn() {
      return request.auth != null;
    }

    function isParticipant() {
      return signedIn() && request.auth.uid in resource.data.participants;
    }

    function isNewParticipant() {
      return signedIn() && request.auth.uid in request.resource.data.participants;
    }

    function isMutualMingl(connectionId) {
      return exists(/databases/$(database)/documents/minglConnections/$(connectionId))
        && get(/databases/$(database)/documents/minglConnections/$(connectionId)).data.status == "mutual"
        && request.auth.uid in get(/databases/$(database)/documents/minglConnections/$(connectionId)).data.participants;
    }

    match /minglConnections/{connectionId} {
      allow read: if isParticipant();
      allow create: if isNewParticipant()
        && request.resource.data.participants.size() == 2
        && request.auth.uid == request.resource.data.requestedBy;
      allow update: if isParticipant()
        && request.resource.data.participants == resource.data.participants
        && (
          request.auth.uid == resource.data.requestedTo ||
          request.auth.uid == resource.data.requestedBy
        );
      allow delete: if false;
    }

    match /chatRooms/{roomId} {
      allow read: if isParticipant();
      allow create: if isNewParticipant()
        && request.resource.data.type == "mingl"
        && isMutualMingl(request.resource.data.connectionId);
      allow update: if isParticipant()
        && resource.data.type == "mingl"
        && isMutualMingl(resource.data.connectionId);
      allow delete: if false;
    }

    match /chatMessages/{messageId} {
      allow read: if signedIn() && request.auth.uid in resource.data.participants;
      allow create: if signedIn()
        && request.auth.uid == request.resource.data.senderUid
        && request.auth.uid in request.resource.data.participants
        && exists(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId))
        && request.auth.uid in get(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId)).data.participants
        && get(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId)).data.type == "mingl"
        && isMutualMingl(get(/databases/$(database)/documents/chatRooms/$(request.resource.data.roomId)).data.connectionId);
      allow update, delete: if false;
    }
  }
}
